# مخططات تدفق البيانات للفصل الثالث

يحتوي هذا الملف على مخططات تدفق البيانات المقترحة بصيغة `Mermaid`، ويمكن نسخ كل مخطط وتحويله إلى صورة لإدراجه في وثيقة المشروع.

## 1. المخطط البيئي

```mermaid
flowchart LR
    %% تعريف تنسيق الألوان
    classDef entity fill:#E1F5FE,stroke:#01579B,stroke-width:2px,color:#000;
    classDef system fill:#E8F5E9,stroke:#2E7D32,stroke-width:3px,color:#000,font-weight:bold;
    classDef external fill:#F3E5F5,stroke:#7B1FA2,stroke-width:2px,color:#000;
    linkStyle default stroke:#455A64,stroke-width:1.5px;

    %% الكيانات الخارجية
    C["العميل"]:::entity
    E["موظف التأمين"]:::entity
    M["الإدارة"]:::entity
    X["جهات خارجية\n(بنوك / معاينة)"]:::external

    %% النظام ككتلة واحدة
    S(["0\nمنصة أمان لإدارة التأمينات"]):::system

    %% تدفق البيانات مع العميل
    C -- "طلب تسجيل / تأمين / رفع مرفقات" --> S
    S -- "وثيقة التأمين / حالة المطالبة / الإشعارات" --> C

    %% تدفق البيانات مع الموظف
    E -- "مراجعة الطلبات / اعتماد / معالجة" --> S
    S -- "قوائم المهام / بيانات العملاء / سجل المتابعة" --> E

    %% تدفق البيانات مع الإدارة
    M -- "السياسات / الصلاحيات / طلب تقارير" --> S
    S -- "التقارير الإحصائية / مؤشرات الأداء / سجل العمليات" --> M

    %% تدفق البيانات مع الجهات الخارجية
    X -- "نتائج عمليات الدفع / تقارير المعاينة" --> S
    S -- "أوامر دفع / طلبات فحص / تقارير دورية" --> X

```

## 2. المخطط الصفري

```mermaid
flowchart TB
    %% تعريف تنسيق الألوان
    classDef entity fill:#E1F5FE,stroke:#01579B,stroke-width:2px,color:#000;
    classDef process fill:#E8F5E9,stroke:#2E7D32,stroke-width:2px,color:#000;
    classDef store fill:#FFFDE7,stroke:#F57F17,stroke-width:2px,color:#000;
    classDef external fill:#F3E5F5,stroke:#7B1FA2,stroke-width:2px,color:#000;
    linkStyle default stroke:#455A64,stroke-width:1px;

    %% الكيانات الخارجية
    C["العميل"]:::entity
    E["موظف التأمين"]:::entity
    M["الإدارة"]:::entity
    X["جهات خارجية\n(بنوك/معاينة)"]:::external

    %% العمليات الرئيسية (النظام)
    subgraph System_Processes [نظام منصة أمان]
        direction TB
        P1(["1.0\nإدارة العملاء والمستخدمين"]):::process
        P2(["2.0\nإدارة الاكتتاب والتسعير"]):::process
        P3(["3.0\nإدارة الوثائق التأمينية"]):::process
        P4(["4.0\nإدارة المطالبات"]):::process
        P5(["5.0\nإدارة الأقساط والمدفوعات"]):::process
        P6(["6.0\nإدارة الإشعارات والرقابة"]):::process
    end

    %% مخازن البيانات
    subgraph Data_Stores [قواعد البيانات]
        direction LR
        D1[("D1\nبيانات العملاء")]:::store
        D2[("D2\nالمنتجات والتسعير")]:::store
        D3[("D3\nالوثائق التأمينية")]:::store
        D4[("D4\nالمطالبات والمرفقات")]:::store
        D5[("D5\nالأقساط والمدفوعات")]:::store
        D6[("D6\nسجل العمليات")]:::store
    end

    %% تدفق البيانات: العميل
    C -- "بيانات التسجيل" --> P1
    C -- "طلب تأمين ومرفقات" --> P2
    C -- "طلب تجديد/تعديل" --> P3
    C -- "بلاغ حادث ومستندات" --> P4
    C -- "طلب سداد قسط" --> P5
    
    P1 -- "تأكيد تنشيط الحساب" --> C
    P3 -- "نسخة الوثيقة المصدرة" --> C
    P4 -- "إشعار قرار المطالبة" --> C
    P5 -- "إيصال سداد معتمد" --> C
    P6 -- "تنبيهات وإشعارات عامة" --> C

    %% تدفق البيانات: الموظف
    E -- "بيانات التوظيف والمهام" --> P1
    E -- "تحديث منتجات وسياسات" --> P2
    E -- "مراجعة الوثائق" --> P3
    E -- "معالجة وتحقيق مطالبات" --> P4
    E -- "متابعة تحصيل مالي" --> P5
    
    P1 -- "صلاحيات وقوائم عمل" --> E
    P2 -- "نتائج التسعير المبدئي" --> E
    P4 -- "قائمة المطالبات الجديدة" --> E
    P6 -- "تنبيهات إدارية" --> E

    %% تدفق البيانات: الإدارة
    M -- "تعيين أدوار وصلاحيات" --> P1
    M -- "اعتماد سياسات التسعير" --> P2
    M -- "طلب تقارير أداء" --> P6
    P6 -- "تقارير تحليلية وKPIs" --> M

    %% تدفق البيانات: الجهات الخارجية
    X -- "تأكيد دفع/تقرير معاينة" --> P4
    X -- "نتائج عمليات الدفع" --> P5
    P4 -- "طلب معاينة فنية" --> X
    P5 -- "أمر دفع إلكتروني" --> X

    %% ربط العمليات بمخازن البيانات (القراءة والكتابة)
    P1 <-->|"بيانات الحسابات"| D1
    P2 <-->|"قواعد التسعير"| D2
    P2 -- "تسجيل عميل جديد" --> D1
    P3 <-->|"تحديث الوثائق"| D3
    P3 -. "تحقق" .-> D1 & D2
    P4 <-->|"ملفات المطالبات"| D4
    P4 -. "تحقق وثيقة" .-> D3
    P5 <-->|"سجلات الدفع"| D5
    P5 -. "تحقق قسط" .-> D3
    P6 <-->|"أرشفة الإشعارات"| D6
    P6 -. "قراءة للتقارير" .-> D1 & D3 & D4 & D5

```
